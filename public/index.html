Integración
3 líneas para empezar
// Crear un asistente con VoiceCore API
const response = await fetch('https://api.voicecore.dev/api/assistants', {
  method: 'POST',
  headers: { 'x-api-key': 'vc_tu_api_key' },
  body: JSON.stringify({
    name: 'Recepcionista Clínica',
    voice: 'sofia-es',
    model: 'groq/llama-3.3-70b',
    ttsProvider: 'cartesia',
    firstMessage: 'Hola, clínica dental. ¿En qué puedo ayudarle?'
  })
});
// ✅ Assistant creado — conecta tu número Twilio y listo
